export const example1 = `
<div class="container">This is some content</div>
`;
export const example1Style = `.container {
  margin-left: var(--container-margin-left, auto);
  margin-right: var(--container-margin-right, auto);
  max-width: var(--container-max-width, 100vw);
  padding-left: var(--container-padding-left, 1ch);
  padding-right: var(--container-padding-right, 1ch);
  width: var(--container-width, 100%);

    &:not(.container--fluid) {
    @media (min-width: 576px) {
      max-width: var(--container-max-width, 516px);
    }

    @media (min-width: 768px) {
      max-width: var(--container-max-width, 708px);
    }

    @media (min-width: 1024px) {
      max-width: var(--container-max-width, 964px);
    }

    @media (min-width: 1216px) {
      max-width: var(--container-max-width, 1156px);
    }

    @media (min-width: 1408px) {
      max-width: var(--container-max-width, 1348px);
    }
  }
}`;

## Fluid container

As you will have seen above, it is possible to add a `.container--fluid` class to your container and it will have a max width of `100vw` regardless of screen size.

export const example2 = `
<div class="container container--fluid">This is some content</div>
`;
